P) Media Queries

						


      						Applications

   .css différent :
   <link rel="stylesheet" media="screen and (max-width: 128px)" type="text/css" href="machin_resolution.css">

   dans le .css actuel :
   @media screen and (max-width: 1280px)
   {

   }

   dans <head></head> :
   <meta name="viewport" content="width=device-width"> (ici viewport = largeur de l'écran)
					
					
						

    	 	   Principales règles

      color (bits/pixels)
      height/width (affichage fenêtre)
      device-height/width (- du périphérique)
      orientation (portrait ou paysage)
      media (écran de sortie)
      	screen (écran classique)
      	handheld (mobile. Préferer max-device width: px)
      	print (impression)
      	tv (tv)
      	projection (projecteur)
      	all (tous type d'écrans)

      -min/-max (sur beaucoup d'entre-eux)
						
					
						

     					only / and / not

	      /* Sur les écrans, quand la largeur de la fenêtre fait au maximum 1280px */
	@media screen and (max-width: 1280px)

	/* Sur tous types d'écran, quand la largeur de la fenêtre est comprise entre 1024px et 1280px */
	@media all and (min-width: 1024px) and (max-width: 1280px)

	/* Sur les téléviseurs */
	@media tv

	/* Sur tous types d'écrans orientés verticalement */
	@media all and (orientation: portrait)
						
					
						

	Exemple de changement

  @media all and (max-width: 10224px)	
  {
	  #bloc_machin
	  {
		  width: auto;
	  }
  }
